<script>
const SESSION_KEY = 'engagex_session';
const GAS_URL = 'https://script.google.com/macros/s/AKfycbxQZju9JeTJMPYqgzSb5D0PwU84EABOTYJNMgBlyGfJ286_U_4UU7HYWLDEaTMyDdg/exec';

function getSession() {
  try { return JSON.parse(localStorage.getItem(SESSION_KEY)) || null; } catch(e) { return null; }
}

function doLogout() {
  localStorage.removeItem(SESSION_KEY);
  window.location.href = 'index.html';
}

function statusClass(s) {
  return 'status-' + (s || 'Pending').replace(/\s+/g, '-');
}

function formatDate(iso) {
  if (!iso) return '—';
  try { return new Date(iso).toLocaleDateString('en-IN', {day:'numeric',month:'short',year:'numeric'}); }
  catch(e) { return iso; }
}

async function loadOrders(email) {
  try {
    const res = await fetch(GAS_URL + '?action=getOrders&email=' + encodeURIComponent(email));
    const text = await res.text();
    const match = text.match(/\{[\s\S]*\}/);
    if (match) {
      const data = JSON.parse(match[0]);
      if (data && data.success && data.orders) return data.orders;
    }
  } catch(e) { console.warn('Orders fetch:', e); }
  const s = getSession();
  return (s && s.orders) ? s.orders : [];
}

function renderOrders(orders) {
  const list = document.getElementById('ordersList');
  const active = orders.filter(o => ['Pending','Verified','In Progress'].includes(o.status || o.Status)).length;
  const done = orders.filter(o => (o.status || o.Status) === 'Completed').length;
  document.getElementById('statTotal').textContent = orders.length;
  document.getElementById('statActive').textContent = active;
  document.getElementById('statDone').textContent = done;

  if (!orders.length) {
    list.innerHTML = `<div class="no-orders">
      <div style="font-size:48px;margin-bottom:12px;">📋</div>
      <h3 style="font-family:'Syne',sans-serif;font-size:20px;font-weight:800;margin-bottom:8px;">No Orders Yet</h3>
      <p style="color:var(--gray);">Place your first order and it will appear here.</p>
      <a href="index.html#pricing" style="display:inline-block;margin-top:20px;padding:12px 28px;background:linear-gradient(135deg,#D4A017,#F5C842);color:#000;border-radius:10px;font-weight:700;text-decoration:none;">Browse Services →</a>
    </div>`;
    return;
  }

  list.innerHTML = orders.slice().reverse().map(o => {
    const id      = o.orderId || o.OrderID || '—';
    const service = o.service || o.Service || '—';
    const biz     = o.bizName || o.BizName || '';
    const total   = o.total   || o.Total   || '—';
    const status  = o.status  || o.Status  || 'Pending';
    const txn      = o.txnId    || o.TxnID    || '';
    const date     = o.submittedAt || o.SubmittedAt || '';
    const progress = parseInt(o.progress || 0);
    const target   = parseInt(o.target   || 0);
    const pct      = target > 0 ? Math.min(100, Math.round((progress / target) * 100)) : 0;
    const note     = o.note || '';
    const showProgress = target > 0;
    return `<div class="order-card">
      <div class="order-top">
        <div>
          <div class="order-id">${id}</div>
          <div class="order-service">${service}</div>
          ${biz ? `<div class="order-biz">${biz}</div>` : ''}
          <div class="order-date">📅 ${formatDate(date)}</div>
        </div>
        <span class="status-badge ${statusClass(status)}">${status}</span>
      </div>
      <div class="order-details">
        <div><div class="d-label">Amount</div><div class="d-val">${total}</div></div>
        ${txn ? `<div><div class="d-label">Txn ID</div><div class="d-val" style="font-family:monospace;font-size:12px;">${txn}</div></div>` : ''}
      </div>
      ${showProgress ? `
      <div style="margin-top:14px;padding:14px;background:rgba(255,255,255,0.02);border-radius:10px;border:1px solid rgba(212,160,23,0.1);">
        <div style="display:flex;justify-content:space-between;font-size:13px;margin-bottom:8px;">
          <span style="color:var(--gray);">Progress</span>
          <span style="color:var(--gold-light);font-weight:700;">${progress} / ${target} · ${pct}%</span>
        </div>
        <div style="background:#111;border-radius:6px;height:10px;overflow:hidden;">
          <div style="height:10px;background:linear-gradient(90deg,#D4A017,#F5C842);width:${pct}%;border-radius:6px;"></div>
        </div>
        ${note ? `<p style="margin-top:10px;font-size:13px;color:rgba(255,255,255,0.6);background:#111;padding:10px 12px;border-radius:8px;border-left:3px solid #F5C842;">📝 ${note}</p>` : ''}
      </div>` : ''}
    </div>`;
  }).join('');
}

async function init() {
  const user = getSession();
  if (!user) {
    document.getElementById('notLoggedIn').classList.add('active');
    return;
  }
  document.getElementById('dashboardScreen').classList.add('active');
  document.getElementById('dashName').textContent = (user.name || '').split(' ')[0] || 'there';
  document.getElementById('dashEmail').textContent = user.email || '';
  if (user.registeredAt) document.getElementById('statSince').textContent = formatDate(user.registeredAt);
  const orders = await loadOrders(user.email);
  renderOrders(orders);
}

function toggleMenu() {
  const nav = document.querySelector('.nav-links');
  if (nav.style.display === 'flex') {
    nav.style.display = 'none';
  } else {
    nav.style.cssText = 'display:flex;flex-direction:column;position:absolute;top:68px;left:0;right:0;background:#0A0A0A;padding:20px 6%;border-bottom:1px solid rgba(255,255,255,0.07);gap:16px;';
  }
}

document.addEventListener('DOMContentLoaded', init);
